let video = document.getElementById("video"); let stream = null; let scanning = false; let currentUser = null; // START CAMERA (FIXED) function startCam(){ // ❗ Stop previous camera if running if(stream){ stream.getTracks().forEach(track => track.stop()); } navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } }) .then(s => { stream = s; video.srcObject = stream; scanning = true; // start scanning after camera starts scanLoop(); }) .catch(()=>alert("Camera error")); } // SCAN LOOP function scanLoop(){ function loop(){ if(scanning) scan(); requestAnimationFrame(loop); } loop(); } // SCAN FUNCTION function scan(){ let canvas = document.getElementById("canvas"); let ctx = canvas.getContext("2d"); canvas.width = video.videoWidth; canvas.height = video.videoHeight; ctx.drawImage(video, 0, 0); let y = canvas.height / 2; let step = canvas.width / 8; let pattern = ""; for(let i=0; i<8; i++){ let d = ctx.getImageData(i*step, y, 1, 1).data; let r = d[0], g = d[1], b = d[2]; pattern += (r > g && r > b) ? "1" : "0"; } // Check against accounts let match = users.find(u => u.pattern === pattern); if(match){ scanning = false; // 🔴 STOP SCANNING stopCamera(); // 🔴 STOP CAMERA currentUser = match; document.getElementById("pinBox").classList.remove("hidden"); } } // STOP CAMERA (IMPORTANT FIX) function stopCamera(){ if(stream){ stream.getTracks().forEach(track => track.stop()); stream = null; } video.srcObject = null; } // PIN CHECK function verifyPin(){ let pin = document.getElementById("pinInput").value; if(currentUser && pin === currentUser.pin){ document.getElementById("pinBox").classList.add("hidden"); document.getElementById("dash").classList.remove("hidden"); document.getElementById("name").innerText = currentUser.name; document.getElementById("info").innerHTML = ` Job: ${currentUser.job}
CNIC: ${currentUser.cnic}
Card: ${currentUser.card}
Balance: ${currentUser.balance} `; }else{ alert("Wrong PIN"); currentUser = null; startCam(); // 🔄 restart camera after failure } } // LOGOUT function logout(){ currentUser = null; document.getElementById("dash").classList.add("hidden"); startCam(); // restart scanning }
